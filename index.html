       </xjc>

task xjcTask << {
    ant.taskdef(name: "xjc", classname: "com.sun.tools.xjc.XJCTask", classpath: configurations.compile.asPath)
    ant.xjc(package: "", destDir: "") {
        schema(dir: "", includes: "")
    }
}
